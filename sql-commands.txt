CREATE TABLE customers (
    email VARCHAR(255) PRIMARY KEY,
    customer_name VARCHAR(255),
    password VARCHAR(255),
    contact_phone VARCHAR(20)
);

CREATE TABLE bookings (
    booking_number VARCHAR(20) PRIMARY KEY,
    customer_email VARCHAR(255),
    passenger_name VARCHAR(255),
    passenger_contact_phone VARCHAR(20),
    pickup_unit_number VARCHAR(20),
    pickup_street_number VARCHAR(20),
    pickup_street_name VARCHAR(255),
    pickup_suburb VARCHAR(100),
    pickup_date DATE,
    pickup_time TIME,
    destination_suburb VARCHAR(100),
    booking_datetime DATETIME DEFAULT CURRENT_TIMESTAMP,
    status VARCHAR(50) DEFAULT 'unassigned',
    FOREIGN KEY (customer_email) REFERENCES customers(email)
);

SET GLOBAL time_zone = '+11:00';
